Code Sample: Robot Vision System

Extract from the Dish-n-Dash project's computer vision and object classification system:


def process_frame(frame):
    """
    Process camera frame for object detection and classification.
    
    Args:
        frame: Current camera frame from RealSense camera
        
    Returns:
        detected_objects: List of objects with their classes and positions
    """
    # Preprocess frame for DETR model
    preprocessed = preprocess_for_detr(frame)
    
    # Run detection model
    with torch.no_grad():
        predictions = model(preprocessed)
    
    # Postprocess results
    detected_objects = []
    for score, label, box in zip(predictions['scores'], predictions['labels'], predictions['boxes']):
        if score > DETECTION_THRESHOLD:
            # Convert box to robot coordinates
            robot_coordinates = camera_to_robot_transform(box)
            
            # Classify as food or utensil
            category = 'food' if label in FOOD_LABELS else 'utensil'
            
            detected_objects.append({
                'label': LABEL_MAP[label],
                'category': category,
                'position': robot_coordinates,
                'confidence': score.item()
            })
    
    return detected_objects
